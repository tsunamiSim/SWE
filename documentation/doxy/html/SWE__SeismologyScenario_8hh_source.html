204  float getBathymetry(float i_time, float x, float y) {
205  int bestXBath, bestYBath;
206  float result;
207  Array::lookUp(y, m_by.getSize(), m_by.elemVector(), &bestYBath);
208  Array::lookUp(x, m_bx.getSize(), m_bx.elemVector(), &bestXBath);
209  result = m_bz[bestYBath][bestXBath];
210  if(!(x < m_disLeft || x > m_disRight || y < m_disBot || y > m_disTop)) {
211  int interIndex = m_dt.getSize() - 2;
212  float interFactor = -1;
213 
214  for(int i = 0; i < m_dt.getSize() - 2; i++)
215  if(m_dt[i + 1] >= i_time) {
216  interIndex = i;
217  break;
218  }
219 
220  if(i_time <= 0)
221  interFactor = 0;
222  else if(i_time >= m_dt[m_dt.getSize() - 1])
223  interFactor = 1;
224  else
225  interFactor = (i_time - m_dt[interIndex]) / (m_dt[interIndex + 1] - m_dt[interIndex]);
226 
227  int bestXDis, bestYDis;
228  Array::lookUp(y, m_dy.getSize(), m_dy.elemVector(), &bestYDis);
229  Array::lookUp(x, m_dx.getSize(), m_dx.elemVector(), &bestXDis);
230 
231  result += m_dz[interIndex][bestYDis][bestXDis] * (1 - interFactor) +
232  m_dz[interIndex + 1][bestYDis][bestXDis] * interFactor;
233 
234  }
235  if(result < 0.f && result > -20.f)
236  return -20.f;
237  else if(result > 0.f && result < 20.f)
238  return 20.f;
239  else
240  return result;
