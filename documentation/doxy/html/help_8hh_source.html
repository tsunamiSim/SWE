215  static Float2D compress(const Float2D &input, int compress, int cutColsLeft, int cutColsRight, int cutRowsBot, int cutRowsTop) {
216 
217  // Create temporary matrix to store compressed data to
218 // std::cout << "writing with compression " << compress << std::endl;
219  Float2D tmp((int) ceil((input.getCols() - cutColsLeft - cutColsRight) / (float) compress), (int) ceil((input.getRows() - cutRowsBot - cutRowsTop) / (float) compress));
220 // std::cout << "matrix cols: " << input.getCols() << ", matrix rows: " << input.getRows() << ", compressed cols: " << tmp.getCols() << ", compressed rows: " << tmp.getRows() << std::endl;
221  // Define variables for building average over one block average
222  float currentValue;
223  int div;
224  // Fill each cell of the compressed matrix
225  for(int _x = 0; _x < tmp.getCols(); _x++) {
226  for(int _y = 0; _y < tmp.getRows(); _y++) {
227 // std::cout << "writing at cell [" << _x << ", " << _y << "]: ";
228  // Initialize average variables
229  currentValue = 0;
230  div = 0;
231  // Iterate over the initial matrix' compression blocks
232  for(int _orig_x = _x * compress;
233  // compression boundary
234  _orig_x < (_x + 1) * compress &&
235  // for the last block make sure to not iterate out of the original matrix
236  _orig_x < input.getCols() - cutColsLeft - cutColsRight;
237  _orig_x++) {
238  for(int _orig_y = _y * compress;
239  // compression boundary
240  _orig_y < (_y + 1) * compress &&
241  // original boundary
242  _orig_y < input.getRows() - cutRowsBot - cutRowsTop;
243  _orig_y++) {
244  // Add the original value to the currentValue buffer and increase the amount of values in this block by one
245  currentValue += input[_orig_x + cutColsLeft][_orig_y + cutRowsBot];
246  div++;
247 // std::cout << "[" << _orig_x + cutColsLeft << ", " <<
248 // _orig_y + cutRowsBot << "]: " <<
249 // input[_orig_x + cutColsLeft][_orig_y + cutRowsBot] << ", ";
250  }
251  }
252 // std::cout << "=> " << currentValue << "/" << div << "=" << (currentValue / div) << std::endl;
253  // write calculated average to compressed matrix
254  tmp[_x][_y] = currentValue / div;
255  }
256  }
257  return tmp;
258  }
